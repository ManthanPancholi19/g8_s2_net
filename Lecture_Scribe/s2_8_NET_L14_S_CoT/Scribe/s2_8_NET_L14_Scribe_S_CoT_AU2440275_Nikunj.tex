\section*{Question 1}

\textbf{Problem:}  
A point is chosen at random on a line segment of length $L$.  
Find the probability that the ratio of the shorter to the longer segment is less than $\frac14$.

\subsection*{Step 1: Interpretation}

Let $X \sim \text{Uniform}(0,L)$.

Segments formed:
\[
X \quad \text{and} \quad L - X
\]

Shorter segment: $\min(X, L-X)$  
Longer segment: $\max(X, L-X)$

\subsection*{Step 2: Inequality}

We require:
\[
\frac{\min(X,L-X)}{\max(X,L-X)} < \frac14
\]

\textbf{Case 1:} $0 < X \le \frac{L}{2}$

\[
\frac{X}{L-X} < \frac14
\]

\[
4X < L - X
\]

\[
5X < L
\]

\[
X < \frac{L}{5}
\]

\textbf{Case 2:} $\frac{L}{2} < X < L$

\[
\frac{L-X}{X} < \frac14
\]

\[
4(L-X) < X
\]

\[
4L - 4X < X
\]

\[
4L < 5X
\]

\[
X > \frac{4L}{5}
\]

\subsection*{Step 3: Probability}

Favorable region:
\[
X < \frac{L}{5}
\quad \text{or} \quad
X > \frac{4L}{5}
\]

Each interval has length $\frac{L}{5}$.

\[
P = \frac{2(L/5)}{L} = \boxed{\frac{2}{5}}
\]

\section*{Question 2}

White region: $N(4,4)$  
Black region: $N(6,9)$  

Let fraction black $= \alpha$.

Equal error condition at $x=5$:

\[
\alpha f_B(5) = (1-\alpha) f_W(5)
\]

Normal density:
\[
f(x)=\frac{1}{\sqrt{2\pi\sigma^2}} e^{-(x-\mu)^2/(2\sigma^2)}
\]

White:
\[
f_W(5)=\frac{1}{\sqrt{2\pi \cdot 4}} e^{-1/(8)}
\]

Black:
\[
f_B(5)=\frac{1}{\sqrt{2\pi \cdot 9}} e^{-1/(18)}
\]

Solving:
\[
\alpha = \frac{f_W(5)}{f_W(5)+f_B(5)}
\]

\section*{Question 3}

\subsection*{(a) Uniform(0,A)}

Minimize $E|X-a|$.

Minimizer = Median.

\[
\boxed{a = \frac{A}{2}}
\]

\subsection*{(b) Exponential($\lambda$)}

CDF:
\[
F(x)=1-e^{-\lambda x}
\]

Median $m$:

\[
1-e^{-\lambda m}=\frac12
\]

\[
e^{-\lambda m}=\frac12
\]

\[
\boxed{m=\frac{\ln 2}{\lambda}}
\]

\section*{Question 4}

Mixture exponential:

\[
P(X>t)=p_1 e^{-\lambda_1 t}+p_2 e^{-\lambda_2 t}
\]

\[
P(X>t+s)=p_1 e^{-\lambda_1(t+s)}+p_2 e^{-\lambda_2(t+s)}
\]

\[
P(X>t+s|X>t)
=
\frac{p_1e^{-\lambda_1(t+s)}+p_2e^{-\lambda_2(t+s)}}
{p_1e^{-\lambda_1 t}+p_2e^{-\lambda_2 t}}
\]

\section*{Question 5}

$X \sim \text{Poisson}(\lambda_1)$  
$Y \sim \text{Poisson}(\lambda_2)$

\[
P(X=k|X+Y=n)
=
\frac{P(X=k,Y=n-k)}{P(X+Y=n)}
\]

Substitute pmfs:

\[
=
\binom{n}{k}
\left(\frac{\lambda_1}{\lambda_1+\lambda_2}\right)^k
\left(\frac{\lambda_2}{\lambda_1+\lambda_2}\right)^{n-k}
\]
