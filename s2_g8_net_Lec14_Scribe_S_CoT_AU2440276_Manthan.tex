\section*{Question 1}
Uniform selection on an interval and probability via geometric length ratio. Handling expressions involving min and max through case analysis.

\textbf{Interpretation}

A point is chosen uniformly on a segment of length $L$.  

Let $0 < x < L$.  

Segments formed: $x$ and $L-x$  

\[
R = \frac{\min(x, L-x)}{\max(x, L-x)}
\]

We compute:
\[
P(R < 1/4)
\]

Since selection is uniform:
\[
P = \frac{\text{length of valid region}}{L}
\]

\textbf{Case Analysis (Split at Midpoint)}

Switch occurs at:
\[
x = \frac{L}{2}
\]

\textbf{Case 1:} $x \le \frac{L}{2}$

\[
\frac{x}{L-x} < \frac{1}{4}
\]

\[
4x < L-x
\]

\[
5x < L
\]

\[
x < \frac{L}{5}
\]

\textbf{Case 2:} $x > \frac{L}{2}$

\[
\frac{L-x}{x} < \frac{1}{4}
\]

\[
4L - 4x < x
\]

\[
5x > 4L
\]

\[
x > \frac{4L}{5}
\]

Valid region:
\[
(0, L/5) \cup (4L/5, L)
\]

Total length:
\[
\frac{L}{5} + \frac{L}{5} = \frac{2L}{5}
\]

Final Probability:
\[
P = \frac{2L/5}{L} = \frac{2}{5}
\]

\section*{Question 2}
Bayesâ€™ theorem with continuous densities.  
Equal posterior probabilities imply equal error.

Given:

\[
X|W \sim N(4,4), \quad \sigma_1 = 2
\]
\[
X|B \sim N(6,9), \quad \sigma_2 = 3
\]
\[
P(B)=\alpha, \quad P(W)=1-\alpha
\]
\[
X=5
\]

Equal error condition:
\[
P(B|X=5)=P(W|X=5)=\frac{1}{2}
\]

Bayes formula:
\[
P(B|X=5)=\frac{\alpha f(5|B)}{\alpha f(5|B)+(1-\alpha)f(5|W)}
\]

Normal density:
\[
f(x)=\frac{1}{\sigma\sqrt{2\pi}} e^{-\frac{(x-\mu)^2}{2\sigma^2}}
\]

\[
f(5|B)=\frac{1}{3\sqrt{2\pi}}e^{-1/18}
\]
\[
f(5|W)=\frac{1}{2\sqrt{2\pi}}e^{-1/8}
\]

Substitution:
\[
\frac{3\alpha}{3\alpha + 2(1-\alpha)e^{-5/72}}=\frac{1}{2}
\]

\[
6\alpha = 3\alpha + 2(1-\alpha)e^{-5/72}
\]

\[
3\alpha = 2(1-\alpha)(0.93)
\]

\[
4.86\alpha = 1.86
\]

\[
\alpha \approx 0.3827
\]

\section*{Question 3}

\subsection*{(a) Uniform Case}

\[
X \sim \text{Uniform}(0,A)
\]

\[
f_X(x)=\frac{1}{A}
\]

Minimize:
\[
E[|X-a|]
\]

\[
E[|X-a|]=\int_0^a (a-x)\frac{1}{A}dx + \int_a^A (x-a)\frac{1}{A}dx
\]

After evaluation:
\[
\frac{2a^2 - 2aA + A^2}{2A}
\]

Differentiate:
\[
\frac{1}{2A}(4a-2A)=0
\]

\[
4a=2A
\]

\[
a=\frac{A}{2}
\]

\subsection*{(b) Exponential Case}

\[
X \sim \text{Exponential}(\lambda)
\]

\[
f(x)=\lambda e^{-\lambda x}
\]

\[
\frac{d}{da}E[|X-a|]=1-2e^{-\lambda a}=0
\]

\[
2e^{-\lambda a}=1
\]

\[
e^{-\lambda a}=\frac{1}{2}
\]

\[
a=\frac{\ln 2}{\lambda}
\]
